void solve() {
    std::cin >> n >> m;
    sum = 0, res = 0;
    std::string s[n];
    int len[n], pref[n];
    for (int i = 0; i < n; i++) {
        std::cin >> s[i];
        len[i] = s[i].length();
    }
    if (n == 1) {
        if (len[0] <= m) std::cout << 1 << std::endl;
        else std::cout << 0 << std::endl;
        return;
    }
    pref[0] = len[0];
    if (pref[0] <= m) res++;
    for (int i = 1; i < n; i++) {
        pref[i] = pref[i - 1] + len[i];
        if (pref[i] <= m) {
            res++;
        } else {
            break;
        }
    }
    std::cout << res << std::endl;
}

void solve() {
    ll a[200020] = {0};
    ll sum = 0, avg = 0;
    std::cin >> n;
    for (int i = 1; i <= n; i++) {
        std::cin >> a[i];
        sum += a[i];
    }
    if (sum % n != 0) {
        std::cout << "NO" << std::endl;
        return;
    } else {
        avg = sum / n;
    }
    for (int i = 2; i <= n - 1; i++) {
        if (a[i - 1] < avg) {
            a[i + 1] -= (avg - a[i - 1]);
            a[i - 1] = avg;
        } else {
            a[i + 1] += (a[i - 1] - avg);
            a[i - 1] = avg;
        }
    }

            std::cout << "NO" << std::endl;
            return;
        }
    }
    std::cout << "YES" << std::endl;
}
